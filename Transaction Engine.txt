// Initial Data (This would ideally come from your database)
let currentBalance = 45250.00;

function processTransaction(type, amount, recipientOrSource) {
    const transactionList = document.getElementById('transaction-list');
    const balanceDisplay = document.getElementById('total-balance-display');
    const date = new Date().toLocaleDateString('en-US', {
        month: 'short',
        day: 'numeric',
        year: 'numeric',
        hour: '2-digit',
        minute: '2-digit'
    });

    amount = parseFloat(amount);

    if (type === 'Debit') {
        if (amount > currentBalance) {
            alert("Insufficient Funds!");
            return;
        }
        currentBalance -= amount;
        var statusClass = 'text-danger';
        var sign = '-';
    } else {
        currentBalance += amount;
        var statusClass = 'text-success';
        var sign = '+';
    }

    // Update the Balance on screen
    balanceDisplay.innerText = `$${currentBalance.toLocaleString(undefined, {minimumFractionDigits: 2})}`;

    // Create the New Row for History
    const newRow = `
        <tr>
            <td>
                <strong>${recipientOrSource}</strong><br>
                <small class="text-muted">${date}</small>
            </td>
            <td class="${statusClass}">${sign}$${amount.toLocaleString()}</td>
            <td>${type === 'Debit' ? 'International Wire' : 'Account Top-up'}</td>
            <td><span class="status-success">Successful</span></td>
            <td><button class="btn btn-sm btn-link text-danger"><i class="fas fa-trash"></i></button></td>
        </tr>
    `;

    // Add to the top of the table
    transactionList.insertAdjacentHTML('afterbegin', newRow);
}
